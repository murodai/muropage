130 subroutine genbase_unlim_sub(rk,nb,s,n,loopless,t1,t2,rkmin,rkmax,tmax,z)
131 implicit none
132 integer(8) n,m,t,rk,tmax,nb,nbmin,nbmax,rkmin,rkmax,i,j,k,l,h,d,d1,d2,u
133 integer(8),parameter:: param = 500
134 integer(8) s(param),ns(param)
135 integer(8) factorial,binom
136 logical(1) z,t1,t2,loopless,c
137 real x
138 !!!!! 与える条件
139 !台集合の元の個数
140 !n =
141 !!!!!!!!!!!!!!!!!!!!!
142 z = .false.
143 m     = 2*n        !部分集合の個数
144 do t = 1, tmax
145  call random_number(x)   !基底の個数を定める。
146  nb = int(x*binom(n,n/2)) + 1
147  if (nb > param) cycle
148  do i = 1, nb   !部分集合を無作為に与える。
149  call random_number(x)
150  s(i) = int(x*(2**n))
151  enddo
152  do i = 1, nb    !同じ集合がある場合はやり直し。
153   do j = 1, nb
154     if (i == j) cycle
155     if (s(i) == s(j)) goto 10
156   enddo
157  enddo
158  if (loopless .eqv. .true.) then !ループがあるかを判定する。
159    u = s(1)
160    do i = 2, nb
161       u = ior(u,s(i))
162    enddo
163    if (u /= 2**n-1) goto 10
164  endif
165  do i = 1, nb    !部分集合の大きさを測る。
166       ns(i) = 0
167  enddo
168  do i = 1, nb
169   do j = 1, n
170          if (iand(s(i),2**(j-1)) == 2**(j-1)) ns(i) = ns(i) + 1
171   enddo
172  enddo
173  do i = 1, nb   !元の個数が一様でないならばやり直し。
174   do j = 1, nb
175      if (ns(i) /= ns(j)) goto 10
176   enddo
177  enddo
178  if ((t1 .eqv. .true.) .and. (ns(1) < rkmin)) goto 10     !基底の最小値より小さいならばやり直し。
179  if ((t2 .eqv. .true.) .and. (ns(1) > rkmax)) goto 10     !基底の最大値より大きいならばやり直し。
180 !基底族の公理を満足するか判定する。
181  do i = 1, nb
182   do j = 1, nb
183   if (i == j) cycle
184   d = ieor(s(i),s(j))        !s(i)とs(j)の対称差を取る。
185   d1 = iand(d,s(i))     !s(i)からs(j)を引く。
186   d2 = iand(d,s(j))     !s(j)からs(i)を引く。
187    do k = 1, n
188    if (iand(d1,2**(k-1)) /= 2**(k-1)) cycle
189    c = .false.
190     do l = 1, n
191      if (iand(d2,2**(l-1)) /= 2**(l-1)) cycle
192      do h = 1, nb
193             if ((s(i) - 2**(k-1)) + 2**(l-1) == s(h)) c = .true.
194      enddo
195     enddo
196         if (c .eqv. .false.) goto 10
197    enddo
198   enddo
199  enddo
200     z = .true.
201     exit
202 10 enddo
203     do i = 1, nb
204       print'(b32.32)',s(i)
205     enddo
206     rk = ns(1)
207     if (z .eqv. .true.) print'(A6,I32)','階数 = ',rk
208 print'(A6,I32)','試行回数 = ',t
209 if (z .eqv. .true.) print'(A6)','成功'
210 if (z .eqv. .false.) print'(A6)','失敗'
211 end subroutine genbase_unlim_sub

214 function binom(n,k)
215 implicit none
216 integer(8) binom,n,k,i
217 if (n-k > k) k = n-k
218 binom = n
219 do i = n-1, k + 1,-1
220 binom = binom * i
221 enddo
222 do i = 1, n-k
223 binom = binom / i
224 enddo
225 end function binom
